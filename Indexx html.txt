<script>
  const steps = document.querySelectorAll('.step');
  let current = 0;

  function nextStep() {
    gsap.to(steps[current], { opacity: 0, y: -30, scale: 0.95, duration: .5 });
    steps[current].classList.add('hiddenStep');
    current++;
    steps[current].classList.remove('hiddenStep');
    gsap.fromTo(steps[current], { opacity: 0, y: 40, scale: 0.95 }, { opacity: 1, y: 0, scale: 1, duration: .6 });
    document.getElementById('progressBar').style.width = ((current) / (steps.length - 1)) * 100 + '%';
  }

  function finale() {
    document.getElementById('finalBtn').disabled = true;
    gsap.to('#finalBtn', { opacity: 0, duration: .5 });

    const text = document.getElementById('finalText');
    text.classList.remove('hidden');
    gsap.to(text, { opacity: 1, y: -10, duration: 1 });

    const duration = 5000;
    const end = Date.now() + duration;

    (function frame() {
      confetti({ particleCount: 6, spread: 70, origin: { x: 0, y: 1 } });
      confetti({ particleCount: 6, spread: 70, origin: { x: 1, y: 1 } });
      if (Date.now() < end) requestAnimationFrame(frame);
    })();
  }

  /* THREE.JS HEARTS */
  const scene = new THREE.Scene();
  const camera = new THREE.PerspectiveCamera(75, innerWidth/innerHeight, 0.1, 1000);
  const renderer = new THREE.WebGLRenderer({ canvas: document.getElementById('bg'), alpha: true });
  renderer.setSize(innerWidth, innerHeight);

  const hearts = [];
  const geometry = new THREE.SphereGeometry(0.3, 32, 32);
  const material = new THREE.MeshStandardMaterial({ color: 0xff4d6d, metalness: .6, roughness: .3 });

  const light = new THREE.PointLight(0xffffff, 1);
  light.position.set(10,10,10);
  scene.add(light);

  for (let i=0;i<22;i++){
    const h = new THREE.Mesh(geometry, material.clone());
    h.position.set((Math.random()-0.5)*10,(Math.random()-0.5)*6,Math.random()*-10);
    hearts.push(h);
    scene.add(h);
  }

  camera.position.z = 5;

  function animate(){
    requestAnimationFrame(animate);
    hearts.forEach(h=>{
      h.position.y += Math.sin(Date.now()*0.001 + h.position.x)*0.0008;
      h.rotation.y += 0.002;
    });
    renderer.render(scene,camera);
  }
  animate();

  window.addEventListener('resize', ()=>{
    camera.aspect = innerWidth/innerHeight;
    camera.updateProjectionMatrix();
    renderer.setSize(innerWidth, innerHeight);
  });
</script>
